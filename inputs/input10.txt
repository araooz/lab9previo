struct Resultado {
    var int total;
    var int mayor;
    var int menor
};

fun int sumaLista(list valores, int n)
    var int i,total;
    i = 0;
    total = 0;
    while i < n do
        total = total + valores[i];
        i = i + 1
    endwhile;
    return(total)
endfun

fun int mayorLista(list valores, int n)
    var int i,mayor;
    i = 1;
    mayor = valores[0];
    while i < n do
        if valores[i] > mayor then
            mayor = valores[i]
        endif;
        i = i + 1
    endwhile;
    return(mayor)
endfun

fun int menorLista(list valores, int n)
    var int i,menor;
    i = 1;
    menor = valores[0];
    while i < n do
        if valores[i] < menor then
            menor = valores[i]
        endif;
        i = i + 1
    endwhile;
    return(menor)
endfun

fun int main()
    var list datos;
    var Resultado r;
    datos = new int {25, 42, 17, 88, 5, 63};
    r = new Resultado {};
    r.total = sumaLista(datos, 6);
    r.mayor = mayorLista(datos, 6);
    r.menor = menorLista(datos, 6);
    print(r.total);
    print(r.mayor);
    print(r.menor);
    return(0)
endfun
